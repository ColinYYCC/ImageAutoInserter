```python
def check_and_add_picture_columns(self):
    """
    检查并添加 Picture 1/2/3 列
    
    逻辑:
    1. 获取表头行
    2. 获取最后一列的列名
    3. 检查最后一列的数据行是否有内容
    4. 根据检查结果决定添加位置
    """
    # 获取表头行
    header_row = self.find_header_row()
    
    # 获取最后一列
    last_column = self.get_last_column()
    
    # 检查最后一列的数据行是否有内容
    has_data = False
    for row in range(header_row + 1, self.max_row + 1):
        cell_value = self.sheet.cell(row=row, column=last_column).value
        if cell_value is not None and str(cell_value).strip() != '':
            has_data = True
            break
    
    # 根据检查结果决定添加位置
    if has_data:
        # 最后一列有数据 → 在下一列添加
        start_column = last_column + 1
    else:
        # 最后一列无数据 → 直接在该列添加
        start_column = last_column
    
    # 添加 Picture 1/2/3 列
    for i in range(3):
        column_name = f"Picture {i+1}"
        self.sheet.cell(row=header_row, column=start_column + i).value = column_name
```
